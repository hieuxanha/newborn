CREATE DATABASE nhasach; USE nhasach; CREATE TABLE sach ( maSach INT AUTOINCREMENT PRIMARY KEY, tenSach VARCHAR(255) NOT NULL, theLoai VARCHAR(100), tacGia VARCHAR(255), nhaXuatBan VARCHAR(255), namXuatBBan INT, soLuong INT, giaBan DECIMAL(10, 2), trangThai VARCHAR(50) AS ( CASE WHEN soLuong < 10 THEN 'Sắp hết' WHEN soLuong >= 10 AND soLuong < 50 THEN 'Còn' ELSE 'Số lượng lớn' END ) STORED );
